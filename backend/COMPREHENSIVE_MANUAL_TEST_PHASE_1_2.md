    print(f"‚úÖ Relationships created: {build_result.relationships_created}")
    print(f"‚úÖ Files processed: {build_result.files_processed}")
    print(f"‚úÖ Build duration: {build_result.build_duration_ms:.2f}ms")
    
    # Verify graph structure
    session = neo4j_conn.get_session()
    try:
        # Count nodes by type
        node_counts = {}
        result = session.run("""
            MATCH (n) 
            RETURN labels(n)[0] as label, count(n) as count
            ORDER BY count DESC
        """)
        
        for record in result:
            node_counts[record["label"]] = record["count"]
        
        print(f"‚úÖ Node distribution:")
        for label, count in node_counts.items():
            print(f"   - {label}: {count}")
        
        # Count relationships by type
        rel_counts = {}
        result = session.run("""
            MATCH ()-[r]->() 
            RETURN type(r) as rel_type, count(r) as count
            ORDER BY count DESC
        """)
        
        for record in result:
            rel_counts[record["rel_type"]] = record["count"]
        
        print(f"‚úÖ Relationship distribution:")
        for rel_type, count in rel_counts.items():
            print(f"   - {rel_type}: {count}")
            
        # Verify expected Spring entities
        spring_entities = session.run("""
            MATCH (c:Class) 
            WHERE c.name CONTAINS "Controller" OR c.name CONTAINS "Service" OR c.name CONTAINS "Repository"
            RETURN c.name as name, c.file_path as file_path
            LIMIT 10
        """)
        
        print(f"‚úÖ Spring components found:")
        for record in spring_entities:
            print(f"   - {record['name']} in {record['file_path']}")
            
    finally:
        session.close()
